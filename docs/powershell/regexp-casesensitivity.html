PowerShell Regular Expression Case Sensitivity

Case insensitive match

$string -match $regex

Case sensitive match

$string -cmatch $regex

Case insensitive replace

$string -replace $regex,$replacement

Case sensitive replace

$string -creplace $regex,$replacement

Case insensitive split

$string -split $regex

Case sensitive split

$string -csplit $regex

Case insensitive Select-String

$something | Select-String $regex

Case sensitive Select-String

$something | Select-String -CaseSensitie $regex
